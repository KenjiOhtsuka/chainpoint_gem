def self.select_nodes(number)
  node_list_uri = URI(NODE_LIST_ENDPOINTS.sample)
  servers = JSON.parse(Net::HTTP.get(node_list_uri)).sample(number)
  servers.map { |server| server['public_uri'] }
end
